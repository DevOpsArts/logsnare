Rule-Based Routing Example 
alerting:
  rules:
    # Critical errors to on-call
    - name: critical-errors
      patterns: ["FATAL", "OOMKilled"]
      email:
        toAddresses: ["oncall@company.com"]
        
    # Security to security team
    - name: security-alerts
      patterns: ["authentication failed"]
      webhooks:
        - type: slack
          url: "https://hooks.slack.com/..."
Installation

Get Started in Minutes

Deploy Logsnare using Helm with just a few commands.

Default Installation (File Storage) 
# Clone the repository
git clone https://github.com/DevOpsArts/logsnare.git
cd logsnare

# Install with Helm
helm install logsnare-engine ./charts/logsnare-engine \
  --namespace logsnare \
  --create-namespace
PostgreSQL Backend 
helm install logsnare-engine ./charts/logsnare-engine \
  --namespace logsnare \
  --create-namespace \
  --set storage.type=postgresql \
  --set connections.postgresql.host=postgresql.database.svc \
  --set connections.postgresql.username=logsnare \
  --set connections.postgresql.password=<YOUR_PASSWORD>
Azure Log Analytics 
helm install logsnare-engine ./charts/logsnare-engine \
  --namespace logsnare \
  --create-namespace \
  --set storage.type=azure \
  --set connections.azure.workspaceId=<WORKSPACE_ID> \
  --set connections.azure.sharedKey=<SHARED_KEY>
High Availability (3 Replicas) 
helm install logsnare-engine ./charts/logsnare-engine \
  --namespace logsnare \
  --create-namespace \
  --set replicaCount=3 \
  --set leaderElection.enabled=true \
  --set storage.type=postgresql \
  --set connections.postgresql.host=postgresql.database.svc \
  --set connections.postgresql.username=logsnare \
  --set connections.postgresql.password=<YOUR_PASSWORD>
